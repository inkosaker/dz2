a = int(input("Введите первое число: "))
b = int(input("Введите второе число: "))
op = input("Введите оператор (+, -, *, /): ")
if op == "+":
    r = a + b
elif op == "-":
    r = a - b
elif op == "*":
    r = a * b
elif op == "/":
    if b == 0:
        print("Если вы хотите делить на ноль, то вам в НИУ ВШЭ!")
        exit()
    r = a / b
else:
    print("Если вы не знаете математических операторов, то вам в НИУ ВШЭ!")
print('Результат:', r)
